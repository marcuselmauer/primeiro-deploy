  // ---------- Criptografia SHA-256 ----------
  function randomSalt() {
    const arr = new Uint8Array(16);
    crypto.getRandomValues(arr);
    return Array.from(arr).map(b => b.toString(16).padStart(2, '0')).join('');
  }

  async function hashPassword(password, salt) {
    const enc = new TextEncoder();
    const data = enc.encode(password + '::' + salt);
    const buf = await crypto.subtle.digest('SHA-256', data);
    return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2, '0')).join('');
  }

      users = uData || [];

      const { data: mData, error: mErr } = await supabase.from('members').select('*');
      if (mErr) throw mErr;
      members = mData || [];

      // 3. Carrega Divisões
      const { data: dData, error: dErr } = await supabase.from('divisoes').select('*');
      if (dErr) throw dErr;
      divisoes = dData || [];

      showToast('Erro de sincronização com o banco de dados');

    const hash = await hashPassword(password, u.salt);

    bootStage = 'app';
    activeTab = 'integrantes';
    divisionFilter = 'all';
    commandFilter = 'all';
    render();
    return null;
  }

    divisoes = divisoes.filter(x => x.id !== id);
    render();
    showToast('Divisão removida');
  }

  // ---------- Gerenciamento de Usuários (Mestre e Regional) ----------
  async function addUser(username, password, role, divisaoId, comandoId) {
    if (!username.trim() || !password) return 'Preencha usuário e senha';
    if (users.some(x => x.username.toLowerCase() === username.trim().toLowerCase())) {
      return 'Já existe um usuário com esse nome';
    }

    if (currentUser.role === 'regional' && role !== 'adm' && role !== 'regional') {
      return 'Você só tem permissão para criar usuários ADM ou Regional';
    }

    const salt = randomSalt();
    const hash = await hashPassword(password, salt);

    const newUser = {
      id: Date.now().toString(),
      username: username.trim(),
      salt,
      hash,
      role: role || 'user',
      divisao: divisaoId || null,
      comando: comandoId || null,
      emailAdm: '',
      emailDiretor: '',
      mustChangePassword: false,
      relatorioConcluido: false,
      relatorioConcluidoEm: null,
      relatorioConcluidoPeriodo: null
    };

    const { error } = await supabase.from('users').insert([newUser]);
    if (error) return 'Erro ao criar usuário no banco';

    users.push(newUser);
    render();
    showToast('Usuário adicionado');
    return null;
  }

  async function toggleRelatorioConcluido() {
    if (currentUser.role !== 'adm') return;
    const u = users.find(x => x.id === currentUser.id);
    if (!u) return;

    let targetStatus = false;
    let targetEm = null;
    let targetPeriodo = null;

    if (!isConcluidoAtual(u)) {
      targetStatus = true;
      targetEm = Date.now();
      targetPeriodo = currentPeriodKey();
    }

    const { error } = await supabase.from('users').update({
      relatorioConcluido: targetStatus,
      relatorioConcluidoEm: targetEm,
      relatorioConcluidoPeriodo: targetPeriodo
    }).eq('id', u.id);

    if (error) { showToast('Erro ao atualizar status'); return; }

    u.relatorioConcluido = targetStatus;
    u.relatorioConcluidoEm = targetEm;
    u.relatorioConcluidoPeriodo = targetPeriodo;

    render();
    showToast(u.relatorioConcluido ? 'Relatório marcado como concluído!' : 'Marcação de concluído removida');
  }

  async function updateUserEmail(id, field, value) {
    const u = users.find(x => x.id === id);
    if (!u || u.role !== 'adm') return;
    const allowed = currentUser.role === 'master' || (currentUser.role === 'regional' && u.comando === currentUser.comando);
    if (!allowed) return;

    const payload = {};
    payload[field] = value.trim();

    const { error } = await supabase.from('users').update(payload).eq('id', id);
    if (error) { showToast('Erro ao salvar e-mail'); return; }

    u[field] = value.trim();
    showToast('E-mail salvo');
  }

  async function removeUser(id) {
    const target = users.find(x => x.id === id);
    if (!target) return;
    if (target.role === 'master') { showToast('O usuário programador não pode ser removido'); return; }
    if (currentUser.role === 'regional') {
      if (target.role === 'regional') { showToast('Só o usuário programador pode remover um Regional'); return; }
      if (target.comando !== currentUser.comando) { showToast('Você só pode remover contas do seu comando'); return; }
    }

    const { error } = await supabase.from('users').delete().eq('id', id);
    if (error) { showToast('Erro ao remover usuário'); return; }

    users = users.filter(x => x.id !== id);
    render();
    showToast('Usuário removido');
  }

  // ---------- Filtros Globais de Integrantes ----------
  function visibleMembers() {
    if (currentUser.role === 'master') {
      let filtered = members;
      if (commandFilter !== 'all') {
        const divsDoCmd = divisoes.filter(d => d.comandoId === commandFilter).map(d => d.id);
        filtered = filtered.filter(m => divsDoCmd.includes(m.divisao));
      }
      if (divisionFilter !== 'all') {
        filtered = filtered.filter(m => m.divisao === divisionFilter);
      }
      return filtered;
    }
    
    if (currentUser.role === 'regional') {
      const divsDoCmd = divisoes.filter(d => d.comandoId === currentUser.comando).map(d => d.id);
      let filtered = members.filter(m => divsDoCmd.includes(m.divisao));
      if (divisionFilter !== 'all') {
        filtered = filtered.filter(m => m.divisao === divisionFilter);
      }
      return filtered;
    }
    
    return members.filter(m => m.divisao === currentUser.divisao);
  }

  async function saveMember(data) {
    if (!data.nome.trim() || !data.inscricao.trim()) {
      showToast('Preencha nome e inscrição');
      return;
    }
    if (!data.divisao) {
      showToast('Selecione uma divisão');
      return;
    }
    if (editingMemberId) {
      const { error } = await supabase.from('members').update(data).eq('id', editingMemberId);
      if (error) { showToast('Erro ao editar no banco de dados'); return; }

      const m = members.find(x => x.id === editingMemberId);
      Object.assign(m, data);
      showToast('Alterações salvas');
    } else {
      const newMember = { id: Date.now().toString(), ...data };
      const { error } = await supabase.from('members').insert([newMember]);
      if (error) { showToast('Erro ao cadastrar no banco de dados'); return; }

      members.push(newMember);
      showToast('Integrante adicionado');
    }
    editingMemberId = null;
    render();
  }

  // ---------- Montagem do Copia e Cola Avançado (WhatsApp) ----------
  function buildReportText() {
    const list = visibleMembers();
    if (list.length === 0) return '';
    
    let tituloContexto = 'GERAL';
    if (currentUser.role === 'regional') tituloContexto = getComandoNome(currentUser.comando).toUpperCase();
    if (currentUser.role === 'adm') tituloContexto = getDivisaoNome(currentUser.divisao).toUpperCase();
    if (currentUser.role === 'master' && commandFilter !== 'all') tituloContexto = getComandoNome(commandFilter).toUpperCase();
    if (divisionFilter !== 'all') tituloContexto = getDivisaoNome(divisionFilter).toUpperCase();

    let relatorio = `📊 *RELATÓRIO DE INADIMPLÊNCIA — ${tituloContexto}*\n`;
    relatorio += `📅 *Data:* ${new Date().toLocaleDateString('pt-BR')}\n`;
    relatorio += `────────────────────\n\n`;

    const corpo = list.map(m => {
      const d = divisoes.find(x => x.id === m.divisao);
      const cmdNome = d ? getComandoNome(d.comandoId) : 'Sem Comando';
      const divNome = d ? d.nome : 'Sem Divisão';
      
      const motivoPrevisao = [m.motivo, m.previsao ? `*Previsão:* ${m.previsao}` : ''].filter(Boolean).join(' — ');
      return `👤 *Nome:* ${m.nome}\n` +
             `🆔 *Inscrição:* ${m.inscricao}\n` +
             `🪖 *Comando:* ${cmdNome} / *Divisão:* ${divNome}\n` +
             `⚠️ *Motivo:* ${motivoPrevisao}`;
    }).join('\n\n────────────────────\n\n');

    return relatorio + corpo;
  }

  // ---------- Lembrete de entrega por e-mail (dias 8, 18 e 29) ----------
  let simularLembrete = false;

  function isReminderDay() {
    return simularLembrete || [9, 19, 30].includes(new Date().getDate());
  }

  // ---------- Período de apuração (reseta o "Concluído" automaticamente) ----------
  function currentPeriodKey() {
    const d = new Date();
    const day = d.getDate();
    let periodIndex = 0;
    if (day <= 9) periodIndex = 0;
    else if (day <= 19) periodIndex = 1;
    else periodIndex = 2;
    return `${d.getFullYear()}-${d.getMonth()}-${periodIndex}`;
  }

  function isConcluidoAtual(u) {
    return !!(u && u.relatorioConcluido && u.relatorioConcluidoPeriodo === currentPeriodKey());
  }

  function buildReportTextForDivisao(divisaoId) {
    const list = members.filter(m => m.divisao === divisaoId);
    if (list.length === 0) return '';
    return list.map(m => {
      const motivoPrevisao = [m.motivo, m.previsao ? `previsão ${m.previsao}` : ''].filter(Boolean).join(' - ');
      return `Nome: ${m.nome}\nInscrição: ${m.inscricao}\nMotivo/Previsão: ${motivoPrevisao}`;
    }).join('\n\n');
  }

  function getReminderTargets() {
    if (currentUser.role === 'master' || currentUser.role === 'regional') {
      const divisoesEscopo = currentUser.role === 'master'
        ? divisoes
        : divisoes.filter(d => d.comandoId === currentUser.comando);
      const targets = [];
      divisoesEscopo.forEach(d => {
        const admsDaDivisao = users.filter(u => u.role === 'adm' && u.divisao === d.id);
        const emails = [];
        admsDaDivisao.forEach(u => {
          if (u.emailAdm) emails.push(u.emailAdm);
          if (u.emailDiretor) emails.push(u.emailDiretor);
        });
        const uniqueEmails = [...new Set(emails)];
        if (uniqueEmails.length > 0) {
          targets.push({ divisaoNome: d.nome, emails: uniqueEmails, text: buildReportTextForDivisao(d.id) });
        }
      });
      return targets;
    }
    const self = users.find(u => u.id === currentUser.id);
    if (!self) return [];
    const emails = [self.emailAdm, self.emailDiretor].filter(Boolean);
    if (emails.length === 0) return [];
    return [{ divisaoNome: getDivisaoNome(currentUser.divisao), emails, text: buildReportTextForDivisao(currentUser.divisao) }];
  }

  function buildMailto(target) {
    const subject = encodeURIComponent('Lembrete de inadimplência - ' + target.divisaoNome);
    const body = encodeURIComponent(target.text || 'Nenhum integrante inadimplente nesta divisão no momento.');
    return `mailto:${target.emails.join(',')}?subject=${subject}&body=${body}`;
  }

  function reminderBannerHtml() {
    if (!isReminderDay()) return '';
    const mainMsg = 'Hoje falta 1 dia para entrega do relatório, por favor preencher as inadimplências o quanto antes';
    const targets = getReminderTargets();
    if (targets.length === 0) {
      return `<div class="panel" style="border-color: var(--amber);"><div class="notice-box">${mainMsg}</div></div>`;
    }
    return `<div class="panel" style="border-color: var(--amber);">
      <h2>${mainMsg}</h2>
      <div style="display:flex; flex-direction:column; gap:10px;">
        ${targets.map(t => `
          <div style="display:flex; justify-content:space-between; align-items:center; gap:10px; flex-wrap:wrap;">
            <span>${escapeHtml(t.divisaoNome)}</span>
            <a class="btn btn-primary" href="${buildMailto(t)}">Abrir e-mail de lembrete</a>
          </div>
        `).join('')}
      </div>
    </div>`;
  }

  function statusEntregaHtml() {
    if (currentUser.role === 'adm') return '';
    const divisoesEscopo = currentUser.role === 'master'
      ? (commandFilter !== 'all' ? divisoes.filter(d => d.comandoId === commandFilter) : divisoes)
      : divisoes.filter(d => d.comandoId === currentUser.comando);
    if (divisoesEscopo.length === 0) return '';
    const rows = divisoesEscopo.map(d => {
      const admU = users.find(u => u.role === 'adm' && u.divisao === d.id);
      const concluded = isConcluidoAtual(admU);
      return `<div class="status-row ${concluded ? 'concluded' : ''}">
        <span>${escapeHtml(d.nome)}</span>
        <span style="color:var(--muted); font-size:11px;">${admU ? escapeHtml(admU.username) : 'sem ADM'}</span>
        <span class="status-tag">${concluded ? '✓ Concluído' : 'Pendente'}</span>
      </div>`;
    }).join('');
    return `<div class="panel"><h2>Status de entrega por divisão</h2>${rows}</div>`;
  }

  async function copyText(text) {
    if (!text) { showToast('Nada para copiar'); return; }
    try {
      await navigator.clipboard.writeText(text);
      showToast('Copiado para o WhatsApp!');
      return;
    } catch (e) { /* tenta método de reserva abaixo */ }
    try {
      const ta = document.createElement('textarea');
      ta.value = text;
      ta.style.position = 'fixed';
      ta.style.left = '-9999px';
      document.body.appendChild(ta);
      ta.focus();
      ta.select();
      const ok = document.execCommand('copy');
      document.body.removeChild(ta);
      showToast(ok ? 'Copiado para o WhatsApp!' : 'Não foi possível copiar');
    } catch (e2) {
      showToast('Não foi possível copiar');
    }
  }
